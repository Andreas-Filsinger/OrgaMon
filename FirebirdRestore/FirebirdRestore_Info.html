Format der ini.Datei

 [System]
 "Eintrag-Numero-beginnend-bei-1" = Win32Mapping,
                                    Hostname des Servers,
                                    SambaMapping,
                                    [SYSDBA Passwort] ,
                                    [RestoreDest]

 Beispiel:

 [System]
 1=I:\,192.168.115.92,/ablage/
 2=C:\,127.000.000.001,C:\
