$VERLAG_R=181

select
 Lager.name,
 Beleg.RID,
 Beleg.zusage,
 beleg.termin,
 beleg.person_r
from
 lager
join
 beleg
on
 beleg.lager_r=lager.rid
where
 lager.verlag_r=$VERLAG_R
order by
 lager.name
