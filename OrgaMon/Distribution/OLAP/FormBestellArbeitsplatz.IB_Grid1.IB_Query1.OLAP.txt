$VERLAG_R=117

SELECT
 BBELEG.RID BelegNo,
 BBELEG.BESTELLT,
 BPOSTEN.MENGE_ERWARTET Anzahl,
 CAST(BPOSTEN.ZUSAGE as DATE) ErwartetAm,
 ARTIKEL.NUMERO HeBuNo,
 COALESCE(ARTIKEL.VERLAGNO,'?') BestellNo,
 COALESCE(AUSGABEART.KUERZEL,'') Art,
 BPOSTEN.ARTIKEL Titel
FROM
 BPOSTEN
join
 BBELEG
ON
 (BBELEG.RID=BPOSTEN.BELEG_R)
left JOIN
 ARTIKEL
ON
 (BPOSTEN.ARTIKEL_R=ARTIKEL.RID)
Left JOIN
 AUSGABEART
ON
 (BPOSTEN.AUSGABEART_R=AUSGABEART.RID)
WHERE
 (BBELEG.PERSON_R=$VERLAG_R) AND
 (BPOSTEN.MENGE_ERWARTET>0)
ORDER BY
 BPOSTEN.ZUSAGE
