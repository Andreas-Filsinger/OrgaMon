
$StartDatum='01.01.2011'
$EndeDatum='01.01.2014'


select 
 GELIEFERT.RID as LIEFER_POSITION,
 GELIEFERT.BELEG_R as BELEG_NUMMER,
 GELIEFERT.POSNO as TEILLIEFERUNG, 
 GELIEFERT.ARTIKEL_R as ARTIKEL_NUMMER,
 GELIEFERT.ARTIKEL as ARTIKEL_BEZEICHNUNG,
 GELIEFERT.MENGE_RECHNUNG as VERKAUFS_MENGE,
 GELIEFERT.PREIS as EINZEL_PREIS_BRUTTO,
 GELIEFERT.NETTO as EINZEL_PREIS_IST_NETTO,
 GELIEFERT.MWST as ENTHALTENE_MWST
from
 GELIEFERT
join
 BELEG
on
 (BELEG.RID=GELIEFERT.BELEG_R) and  
 (BELEG.RECHNUNG between $StartDatum and $EndeDatum)   
order by
 GELIEFERT.RID

select
 BELEG.RID as BELEG_NUMMER,
 BELEG.TEILLIEFERUNG as TEILLIEFERUNG,
 BELEG.RECHNUNG as BUCHUNGS_MOMENT,
 BELEG.RECHNUNGS_BETRAG as GESAMTSUMME_BRUTTO,
 BELEG.MOTIVATION as KASSE_GEGEBEN
from
 BELEG
where
 (BELEG.RECHNUNG between $StartDatum and $EndeDatum)   
order by
 BELEG.RID
