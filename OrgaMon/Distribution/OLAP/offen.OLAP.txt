

$Datum_von='19.11.2008'
$Datum_bis='20.11.2008'

select 
 BELEG.PERSON_R,
 PERSON.KONTO_AR,
 VERSAND.BELEG_R,
 VERSAND.TEILLIEFERUNG,
 VERSAND.RECHNUNG,
 VERSAND.AUSGANG,
 VERSAND.LIEFERBETRAG,
 ANSCHRIFT.UST_ID,
 LAND.ISO_KURZZEICHEN as LAND,
 LAND.MWST as EU,
 (select SUM(BETRAG) from AUSGANGSRECHNUNG where VERSAND.BELEG_R=BELEG_R and VERSAND.TEILLIEFERUNG=TEILLIEFERUNG) as SALDO
from 
 VERSAND
join 
 BELEG
on
 (VERSAND.BELEG_R=BELEG.RID)
join
 PERSON
on
 (PERSON.RID=BELEG.PERSON_R)  
join
 ANSCHRIFT
on
 (ANSCHRIFT.RID=PERSON.PRIV_ANSCHRIFT_R)
join 
 LAND
on
 (LAND.RID=ANSCHRIFT.LAND_R)       
where 
 (VERSAND.AUSGANG > $Datum_von) and 
 (VERSAND.AUSGANG < $Datum_bis) and 
 (VERSAND.LIEFERBETRAG is not null) and
 (VERSAND.LIEFERBETRAG<>0.0) and
 ((select SUM(BETRAG) from AUSGANGSRECHNUNG where VERSAND.BELEG_R=BELEG_R and VERSAND.TEILLIEFERUNG=TEILLIEFERUNG)<>0) 
order by
 VERSAND.AUSGANG 

complete

Beleg(BELEG_R,TEILLIEFERUNG) BELEG;NETTO1;SATZ1;STEUER1;NETTO2;SATZ2;STEUER2;NETTO3;SATZ3;STEUER3
Person(PERSON_R) DEBITOR
